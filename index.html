    // Filtrar módulos

    function filterModules(filter) {

      const disciplineSummary = document.getElementById('discipline-summary');

      const moduleGroups = document.querySelectorAll('.module-group');

      const progressSection = document.querySelector('.progress-section');

      disciplineSummary.style.display = 'grid';

      moduleGroups.forEach(group => group.style.display = 'block');

      if (progressSection) progressSection.style.display = 'block';

      switch (filter) {

        case 'todos':

          break;

        case 'concluidos':

          disciplineSummary.style.display = 'none';

          moduleGroups.forEach(group => group.style.display = 'none');

          break;

        case 'atraso':

          disciplineSummary.style.display = 'none';

          if (progressSection) progressSection.style.display = 'none';

          break;

        case 'disciplina':

          moduleGroups.forEach(group => group.style.display = 'none');

          if (progressSection) progressSection.style.display = 'none';

          break;

      }

    }

    // Gerar calendário

    function generateCalendar() {

      const calendar = document.getElementById('calendar-grid');

      calendar.innerHTML = '';

      const dayHeaders = ['Dom', 'Seg', 'Ter', 'Qua', 'Qui', 'Sex', 'Sáb'];

      dayHeaders.forEach(day => {

        const header = document.createElement('div');

        header.className = 'calendar-header';

        header.textContent = day;

        calendar.appendChild(header);

      });

      const examDays = [18, 21];

      const today = 18;

      const firstDayOfWeek = 2;

      for (let i = 0; i < firstDayOfWeek; i++) {

        const emptyDay = document.createElement('div');

        emptyDay.className = 'calendar-day other-month';

        calendar.appendChild(emptyDay);

      }

      for (let day = 1; day <= 31; day++) {

        const dayElement = document.createElement('div');

        dayElement.className = 'calendar-day';

        dayElement.innerHTML = `<span>${day}</span>`;

        if (day === today) {

          dayElement.classList.add('today');

        }

        if (examDays.includes(day)) {

          dayElement.classList.add('has-exam');

          dayElement.innerHTML += '<div class="exam-indicator"></div>';

          const examTitle = day === 18 ?

            'DIC - Futebol Metodologia (15:00)' :

            'DIC - Futebol Esforço (10:30)';

          dayElement.title = examTitle;

        }

        calendar.appendChild(dayElement);

      }

    }

    function confirmInscription(subject, moduleCode, fee) {

      const epoca = document.querySelector('input[name="epoca"]:checked').value;

      if (appState.balance < fee) {

        showNotification(`❌ Saldo insuficiente no cartão de aluno! Necessário: €${fee},00. Carregue o cartão na secretaria.`, 'error');

        return;

      }

      showNotification('🔄 Processando inscrição...', 'info');

      setTimeout(() => {

        appState.balance -= fee;

        document.getElementById('card-balance').textContent = `€${appState.balance.toFixed(2)}`;

        showNotification(`✅ Inscrição confirmada! ${subject} (${epoca}.ª época) enviada para aprovação da DT.`, 'success');

        setTimeout(() => {

          showNotification(`📧 Diretora de Turma notificada. Receberá confirmação quando a inscrição for aprovada.`, 'info');

        }, 3000);

      }, 2000);

    }
